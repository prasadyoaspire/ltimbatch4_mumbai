package com.examly;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.util.Scanner;

public class OnlineMovieTicketBooking {
    public static void main(String[] args) {
        try (Connection conn = getConnection(); Scanner sc = new Scanner(System.in)) {
            processMenu(conn, sc);
        } catch (Exception e) {
            System.out.println("Error: " + e.getMessage());
        }
    }

    public static Connection getConnection() {
        try {
            String url = "jdbc:mysql://localhost/appdb";
            String user = "root";
            String pass = "examly";
            return DriverManager.getConnection(url, user, pass);
        } catch (SQLException e) {
            throw new RuntimeException("Database connection failed", e);
        }
    }

    public static void insertShow(Connection conn, Scanner sc) {
        try {
            int showId = Integer.parseInt(sc.nextLine());
            String movieTitle = sc.nextLine();
            String showDate = sc.nextLine();
            String theaterName = sc.nextLine();
            int availableSeats = Integer.parseInt(sc.nextLine());

            String sql = "INSERT INTO movie_show (show_id, movie_title, show_date, theater_name, available_seats) VALUES (?,?,?,?,?)";
            try (PreparedStatement ps = conn.prepareStatement(sql)) {
                ps.setInt(1, showId);
                ps.setString(2, movieTitle);
                ps.setString(3, showDate);
                ps.setString(4, theaterName);
                ps.setInt(5, availableSeats);
                ps.executeUpdate();
            }

            System.out.println("Movie show added successfully!");
            System.out.println("ShowID: " + showId);
            System.out.println("Movie: " + movieTitle);
        } catch (Exception e) {
            System.out.println("Error inserting show: " + e.getMessage());
        }
    }

    public static void updateSeats(Connection conn, Scanner sc) {
        try {
            int showId = Integer.parseInt(sc.nextLine());
            String movieTitle = sc.nextLine();

            String selectSQL = "SELECT available_seats FROM movie_show WHERE show_id = ? AND movie_title = ?";
            try (PreparedStatement selectPs = conn.prepareStatement(selectSQL)) {
                selectPs.setInt(1, showId);
                selectPs.setString(2, movieTitle);
                try (ResultSet rs = selectPs.executeQuery()) {
                    if (rs.next()) {
                        int newSeats = rs.getInt(1) + 20; // add 20 seats
                        String updateSQL = "UPDATE movie_show SET available_seats = ? WHERE show_id = ?";
                        try (PreparedStatement updatePs = conn.prepareStatement(updateSQL)) {
                            updatePs.setInt(1, newSeats);
                            updatePs.setInt(2, showId);
                            updatePs.executeUpdate();
                        }
                        System.out.println("Seats updated successfully!");
                    } else {
                        System.out.println("Show not found!");
                    }
                }
            }
        } catch (Exception e) {
            System.out.println("Error updating seats: " + e.getMessage());
        }
    }

    public static void deleteShow(Connection conn, Scanner sc) {
        try {
            int showId = Integer.parseInt(sc.nextLine());
            String movieTitle = sc.nextLine();

            String sql = "DELETE FROM movie_show WHERE show_id = ? AND movie_title = ?";
            try (PreparedStatement ps = conn.prepareStatement(sql)) {
                ps.setInt(1, showId);
                ps.setString(2, movieTitle);
                int rows = ps.executeUpdate();
                if (rows > 0) {
                    System.out.println("Movie show deleted successfully!");
                } else {
                    System.out.println("Show not found!");
                }
            }
        } catch (Exception e) {
            System.out.println("Error deleting show: " + e.getMessage());
        }
    }

    public static void displayAllShows(Connection conn) {
        try {
            String sql = "SELECT * FROM movie_show ORDER BY available_seats DESC";
            try (PreparedStatement ps = conn.prepareStatement(sql);
                    ResultSet rs = ps.executeQuery()) {
                while (rs.next()) {
                    System.out.println("ShowID: " + rs.getInt("show_id"));
                    System.out.println("Movie: " + rs.getString("movie_title"));
                    System.out.println("Date: " + rs.getString("show_date"));
                    System.out.println("Theater: " + rs.getString("theater_name"));
                    System.out.println("Available Seats: " + rs.getInt("available_seats"));
                }
            }
        } catch (Exception e) {
            System.out.println("Error displaying shows: " + e.getMessage());
        }
    }

    public static void processMenu(Connection conn, Scanner sc) {
        int choice = Integer.parseInt(sc.nextLine());
        switch (choice) {
            case 1:
                insertShow(conn, sc);
                break;
            case 2:
                updateSeats(conn, sc);
                break;
            case 3:
                deleteShow(conn, sc);
                break;
            case 4:
                displayAllShows(conn);
                break;
            default:
                System.out.println("Invalid Choice");
        }
    }

}
