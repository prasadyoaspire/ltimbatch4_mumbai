package com.examly;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.util.Scanner;

public class OnlineFitnessClassManagement {

    public static void main(String[] args) {
        try (Connection conn = getConnection(); Scanner sc = new Scanner(System.in)) {
            processMenu(conn, sc);
        } catch (Exception e) {
            System.out.println("Error: " + e.getMessage());
        }
    }

    public static Connection getConnection() {
        try {
            String url = "jdbc:mysql://localhost/appdb";
            String user = "root";
            String pass = "examly";
            return DriverManager.getConnection(url, user, pass);
        } catch (SQLException e) {
            throw new RuntimeException("Database connection failed", e);
        }
    }

    public static void insertClass(Connection conn, Scanner sc) {
        try {
            int classId = Integer.parseInt(sc.nextLine());
            String classType = sc.nextLine();
            String classDate = sc.nextLine();
            String instructor = sc.nextLine();
            int duration = Integer.parseInt(sc.nextLine());

            String sql = "INSERT INTO fitness_class (class_id, class_type, class_date, instructor, duration_mins) VALUES (?,?,?,?,?)";
            try (PreparedStatement ps = conn.prepareStatement(sql)) {
                ps.setInt(1, classId);
                ps.setString(2, classType);
                ps.setString(3, classDate);
                ps.setString(4, instructor);
                ps.setInt(5, duration);
                ps.executeUpdate();
            }

            System.out.println("Fitness class added successfully!");
        } catch (Exception e) {
            System.out.println("Error inserting class: " + e.getMessage());
        }
    }

    public static void updateClassDuration(Connection conn, Scanner sc) {
        try {
            int classId = Integer.parseInt(sc.nextLine());
            String classType = sc.nextLine();

            String selectSQL = "SELECT duration_mins FROM fitness_class WHERE class_id = ? AND class_type = ?";
            try (PreparedStatement selectPs = conn.prepareStatement(selectSQL)) {
                selectPs.setInt(1, classId);
                selectPs.setString(2, classType);
                try (ResultSet rs = selectPs.executeQuery()) {
                    if (rs.next()) {
                        int newDuration = rs.getInt(1) + 30; // add 30 mins
                        String updateSQL = "UPDATE fitness_class SET duration_mins = ? WHERE class_id = ?";
                        try (PreparedStatement updatePs = conn.prepareStatement(updateSQL)) {
                            updatePs.setInt(1, newDuration);
                            updatePs.setInt(2, classId);
                            updatePs.executeUpdate();
                        }
                        System.out.println("Class duration updated successfully!");
                    } else {
                        System.out.println("Class not found!");
                    }
                }
            }
        } catch (Exception e) {
            System.out.println("Error updating class: " + e.getMessage());
        }
    }

    public static void deleteClass(Connection conn, Scanner sc) {
        try {
            int classId = sc.nextInt();

            String sql = "DELETE FROM fitness_class WHERE class_id = ?";
            try (PreparedStatement ps = conn.prepareStatement(sql)) {
                ps.setInt(1, classId);
                int rows = ps.executeUpdate();
                if (rows > 0) {
                    System.out.println("Fitness class deleted successfully!");
                } else {
                    System.out.println("Class not found!");
                }
            }
        } catch (Exception e) {
            System.out.println("Error deleting class: " + e.getMessage());
        }
    }

    public static void displayAllClasses(Connection conn) {
        try {
            String sql = "SELECT * FROM fitness_class";
            try (PreparedStatement ps = conn.prepareStatement(sql);
                    ResultSet rs = ps.executeQuery()) {
                while (rs.next()) {
                    System.out.println("ClassID: " + rs.getInt("class_id"));
                    System.out.println("Type: " + rs.getString("class_type"));
                    System.out.println("Date: " + rs.getString("class_date"));
                    System.out.println("Instructor: " + rs.getString("instructor"));
                    System.out.println("Duration: " + rs.getInt("duration_mins") + " mins");
                }
            }
        } catch (Exception e) {
            System.out.println("Error displaying classes: " + e.getMessage());
        }
    }

    public static void processMenu(Connection conn, Scanner sc) {
        int choice = Integer.parseInt(sc.nextLine());
        switch (choice) {
            case 1:
                insertClass(conn, sc);
                break;
            case 2:
                updateClassDuration(conn, sc);
                break;
            case 3:
                deleteClass(conn, sc);
                break;
            case 4:
                displayAllClasses(conn);
                break;
            default:
                System.out.println("Invalid Choice");
        }
    }
}
